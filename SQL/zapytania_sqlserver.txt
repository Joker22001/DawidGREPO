/*
2. Wyświetl wszystkie produkty koloru czerwonego i niebieskiego, których nazwa zaczyna się 
na 'L', posortowane wg rozmiaru malejąco i ceny rosnąco. 11*/

5. Wyświetl zamówienia przypisane do obszarów o ID 7,8,9. 12138*/
 	select *
	from sales.SalesOrderHeader SOH join sales.SalesTerritory SST on SOH.TerritoryID = SST.TerritoryID
	where SOH.TerritoryID in (7,8,9)
 
/*
6. Wyświetl 10 ostatnich zamówień,
przypisanych do obszarów o ID 7,8,9 o 
wartości (kolumna SubTotal) mniejszej niż 100. 10*/
 	select top 10 *
	from sales.SalesOrderHeader SOH join sales.SalesTerritory SST on SOH.TerritoryID = SST.TerritoryID
	where SOH.TerritoryID in (7,8,9)
		and SubTotal < 100
